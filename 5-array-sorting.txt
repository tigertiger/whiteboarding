function sorter(array) {
  let sortedArray = [];
  for (let i = 0; i < array.length; i ++ ) {
    if (array[i] <= array[i + 1]) {
      sortedArray.push(array[i]);
    } else if (array === array[i]) {
      sortedArray.push(array[i]);
    }
  }
  return sortedArray;
}
